42/*
43 * COLOR(hexnum)
44 *
45 * it is used to generate SDL_Color from a hex number (e.g., 0xaaff12ff), so
46 * instead of writing
47 *
48 * { 0x44, 0x11, 0x22, 0x33 }
49 *
50 * we simply write build_sdlcolor(0x44112233). It can be provide as a string
51 * as well, i.e, build_sdlcolor("0x44112233") which is useful when providing
52 * colors in JSON files (JSON does not accept hex)
53 */
54inline SDL_Color build_sdlcolor(unsigned long num) {
55 return
56 {
57 static_cast<Uint8>((num >> 24) & 0xff),
58 static_cast<Uint8>((num >> 16) & 0xff),
59 static_cast<Uint8>((num >> 8) & 0xff),
60 static_cast<Uint8>(num & 0xff)
61 };
62}
63
64inline SDL_Color build_sdlcolor(std::string strnum) {
65 return build_sdlcolor(stoul(strnum.erase(0, 2), nullptr, 16));
66}

43 // just a wrapper for accessing a map, just to protect the table,
44 // and to throw a meaning full exception when the key is not found
45 // in the original map
46 //
47 template<typename T>
48 class map_access_wrapper {
49 public:
50 sdl_resource_table<T>& map_;
51 std::string desc_;
52 map_access_wrapper(sdl_resource_table<T>& map, std::string desc) :
53 map_(map), desc_(desc) {
54 }

56 inline auto& at(std::string key) {

60 catch (std::out_of_range& e) {
61 throw "Key '" + key + "' does not exists in '" + //
62 desc_ + "'";
63 }
64 catch (std::exception& e) {
65 throw "Error when accessing key '" + key + "' of '" + //
66 desc_ + "'. Original error was: " + e.what();

121 inline void toggleFullScreen(SCREEN_MODES s) {
122 auto flags = SDL_GetWindowFlags(window_);
123 currentScreenMode_ = s;
124 if (flags & SDL_WINDOW_FULLSCREEN && s == WINDOWED) {
125 SDL_SetWindowFullscreen(window_, 0);
126 }
127 else if (s == FULL_DISPLAY) {
